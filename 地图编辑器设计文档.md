# Unity 地形地图编辑器 - 完整设计文档

## 概述

基于 **Scene 视图** 的网格地图编辑器插件，支持地形可视化绘制（刷子/橡皮）、选中地形预览、实时编辑、保存到 JSON 以及从 JSON 加载恢复编辑。运行时通过静态查询类直接按 ID 读取地形属性，实现通行判断、属性加成（攻击、速度、阻挡等）。

### 文档包含内容

| 章节 | 内容 |
|------|------|
| 基础实现 | 核心代码（5 个文件）+ 操作文档 + 功能清单 |
| 扩展功能 | 6 项完整扩展：运行时地图生成、多层地图、A* 寻路、拖拽绘制、Undo/Redo、悬停预览 |

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 地形资源配置表（ScriptableObject），统一管理所有地形类型。
/// 右键菜单创建：Assets → Create → 地图编辑器 → 地形资源配置表
/// </summary>
[CreateAssetMenu(fileName = "MapResourceConfig", menuName = "地图编辑器/地形资源配置表")]
public class MapResourceConfig : ScriptableObject
{
    public List<TerrainResource> terrainResources = new List<TerrainResource>();

    /// <summary>通过 ID 获取地形资源</summary>
    public TerrainResource GetTerrainById(string terrainId)
    {
        if (string.IsNullOrEmpty(terrainId)) return null;
        foreach (var res in terrainResources)
        {
            if (res != null && res.terrainId.Trim() == terrainId.Trim())
                return res;
        }
        return null;
    }

    /// <summary>检查地形 ID 是否已存在</summary>
    public bool ContainsId(string terrainId)
    {
        return GetTerrainById(terrainId) != null;
    }
}

/// <summary>单个地形资源的完整定义</summary>
[System.Serializable]
public class TerrainResource
{
    [Header("基础信息")]
    public string terrainId;       // 唯一 ID
    public string terrainName;     // 显示名称
    public Sprite terrainSprite;   // 预览图标（可选）

    [Header("通行规则")]
    public DefaultTerrainType defaultType;

    [Header("指定阻挡的单位 ID 列表")]
    public List<string> blockUnitIds = new List<string>();

    [Header("自定义属性列表")]
    public List<CustomProperty> customProperties = new List<CustomProperty>();

    // —— 便捷取值方法 ——

    public int GetInt(string propName)
    {
        var p = FindProperty(propName, PropertyType.Int);
        return p != null ? p.intValue : 0;
    }

    public float GetFloat(string propName)
    {
        var p = FindProperty(propName, PropertyType.Float);
        return p != null ? p.floatValue : 0f;
    }

    public bool GetBool(string propName)
    {
        var p = FindProperty(propName, PropertyType.Bool);
        return p != null && p.boolValue;
    }

    private CustomProperty FindProperty(string name, PropertyType type)
    {
        foreach (var p in customProperties)
        {
            if (p != null && p.propertyName == name && p.propertyType == type)
                return p;
        }
        return null;
    }
}

/// <summary>地形通行规则类型</summary>
public enum DefaultTerrainType
{
    Passable,         // 全体可通过
    Impassable,       // 全体不可通过
    UnitImpassable    // 指定单位不可通过（配合 blockUnitIds 使用）
}

/// <summary>自定义属性条目</summary>
[System.Serializable]
public class CustomProperty
{
    public string propertyName;       // 属性名（如"攻击加成"）
    public PropertyType propertyType; // 属性类型

    // 根据类型取其一
    public int intValue;
    public float floatValue;
    public bool boolValue;

    [TextArea(2, 4)]
    public string propertyEffect;     // 效果描述（策划可读）
}

public enum PropertyType
{
    Int,
    Float,
    Bool
}
```

### 3. GameMapGlobal.cs — 全局地形查询

路径：`Assets/Scripts/Map/GameMapGlobal.cs`

```csharp
/// <summary>
/// 运行时全局地形查询（静态类），任何脚本直接调用。
/// 游戏启动时需先赋值 MapConfig。
/// </summary>
public static class GameMapGlobal
{
    /// <summary>地形配置引用（游戏启动时赋值）</summary>
    public static MapResourceConfig MapConfig;

    /// <summary>通过 ID 获取地形资源</summary>
    public static TerrainResource GetTerrain(string terrainId)
    {
        if (MapConfig == null)
        {
            UnityEngine.Debug.LogError("[GameMapGlobal] MapConfig 未初始化，请先赋值！");
            return null;
        }
        return MapConfig.GetTerrainById(terrainId);
    }

    /// <summary>判断某单位是否能通过指定地形</summary>
    /// <param name="terrainId">地形 ID</param>
    /// <param name="unitId">单位 ID</param>
    /// <returns>true = 可通过，false = 不可通过</returns>
    public static bool CheckUnitCanPass(string terrainId, string unitId)
    {
        var terrain = GetTerrain(terrainId);
        if (terrain == null) return true; // 未知地形默认放行

        switch (terrain.defaultType)
        {
            case DefaultTerrainType.Passable:
                return true;

            case DefaultTerrainType.Impassable:
                return false;

            case DefaultTerrainType.UnitImpassable:
                if (terrain.blockUnitIds == null) return true;
                return !terrain.blockUnitIds.Contains(unitId);

            default:
                return true;
        }
    }
}
```

```csharp
            Debug.Log($"[单位] {unitId} 无法通过地形: {terrainId}");
            return;
```

### 5. MapEditorWindow.cs — 编辑器核心窗口

路径：`Assets/Editor/MapEditor/MapEditorWindow.cs`

```csharp
using System.Collections.Generic;
using System.IO;
using UnityEditor;
using UnityEngine;

/// <summary>
/// 地形地图编辑器主窗口。
/// 提供网格可视化、刷子/橡皮绘制、JSON 导入导出。
/// </summary>
public class MapEditorWindow : EditorWindow
{
    // —— 配置引用 ——
    private MapResourceConfig _config;

    // —— 地图参数 ——
    private string _mapName = "NewMap";
    private int _mapWidth = 30;
    private int _mapHeight = 30;
    private float _cellSize = 1f;

    // —— 编辑状态 ——
    private enum EditMode { Brush, Eraser }
    private EditMode _editMode = EditMode.Brush;
    private int _selectTerrainIndex = 0;

    // —— 地图数据（格子坐标 → 地形 ID） ——
    private Dictionary<Vector2Int, string> _mapGrid = new Dictionary<Vector2Int, string>();

    // —— 滚动 ——
    private Vector2 _scrollPos;

    [MenuItem("地图编辑器/打开地图编辑器")]
    public static void OpenMapEditor()
    {
        var window = GetWindow<MapEditorWindow>("地形地图编辑器");
        window.minSize = new Vector2(300, 400);
    }

    private void OnEnable()
    {
        SceneView.duringSceneGui += OnSceneGUI;
    }

    private void OnDisable()
    {
        SceneView.duringSceneGui -= OnSceneGUI;
    }

    // ==================== 主 OnGUI ====================

    private void OnGUI()
    {
        GUILayout.Space(10);
        EditorGUILayout.LabelField("==== 地形地图编辑器 ====", EditorStyles.boldLabel);

        _config = EditorGUILayout.ObjectField(
            "地形资源配置表", _config, typeof(MapResourceConfig), false) as MapResourceConfig;

        if (_config == null)
        {
            EditorGUILayout.HelpBox(
                "请先创建地形资源配置表：右键 Project → Create → 地图编辑器 → 地形资源配置表",
                MessageType.Warning);
            return;
        }

        DrawMapSetting();
        DrawEditMode();
        DrawTerrainSelect();
        DrawOperateBtn();
        DrawTerrainConfigList();
    }

    // —— 地图参数区域 ——
    private void DrawMapSetting()
    {
        GUILayout.Space(5);
        EditorGUILayout.LabelField("地图基础参数", EditorStyles.boldLabel);
        _mapName = EditorGUILayout.TextField("地图名称", _mapName);
        _mapWidth = Mathf.Max(1, EditorGUILayout.IntField("地图宽度（格子）", _mapWidth));
        _mapHeight = Mathf.Max(1, EditorGUILayout.IntField("地图高度（格子）", _mapHeight));
        _cellSize = Mathf.Max(0.1f, EditorGUILayout.FloatField("格子尺寸", _cellSize));

        EditorGUILayout.HelpBox(
            $"当前地图: {_mapWidth}×{_mapHeight}, 已放置 {_mapGrid.Count} 个地形格子",
            MessageType.Info);

        GUILayout.Space(5);
    }

    // —— 编辑模式 ——
    private void DrawEditMode()
    {
        EditorGUILayout.LabelField("编辑模式", EditorStyles.boldLabel);
        _editMode = (EditMode)EditorGUILayout.EnumPopup("当前模式", _editMode);
    }

    // —— 地形选择 ——
    private void DrawTerrainSelect()
    {
        if (_config.terrainResources.Count == 0)
        {
            EditorGUILayout.HelpBox("地形资源列表为空，请先添加地形", MessageType.Warning);
            return;
        }

        EditorGUILayout.LabelField("选择要绘制的地形", EditorStyles.boldLabel);
        _selectTerrainIndex = Mathf.Clamp(_selectTerrainIndex, 0, _config.terrainResources.Count - 1);
        _selectTerrainIndex = EditorGUILayout.IntSlider(
            "地形下标", _selectTerrainIndex, 0, _config.terrainResources.Count - 1);

        var selected = _config.terrainResources[_selectTerrainIndex];
        EditorGUILayout.LabelField(
            "当前选中地形",
            $"ID: {selected.terrainId} | 名称: {selected.terrainName} | 通行: {selected.defaultType}");
    }

    // —— 操作按钮 ——
    private void DrawOperateBtn()
    {
        GUILayout.Space(5);

        EditorGUILayout.BeginHorizontal();
        if (GUILayout.Button("清空全部地图", GUILayout.Height(30)))
            _mapGrid.Clear();
        if (GUILayout.Button("保存地图为 JSON", GUILayout.Height(30)))
            SaveMapToJson();
        if (GUILayout.Button("从 JSON 加载地图", GUILayout.Height(30)))
            LoadMapFromJson();
        EditorGUILayout.EndHorizontal();

        GUILayout.Space(5);
    }

    // —— 地形配置列表 ——
    private void DrawTerrainConfigList()
    {
        _scrollPos = EditorGUILayout.BeginScrollView(_scrollPos);
        EditorGUILayout.LabelField("地形资源配置列表", EditorStyles.boldLabel);

        if (GUILayout.Button("+ 添加新地形"))
        {
            _config.terrainResources.Add(new TerrainResource());
            EditorUtility.SetDirty(_config);
        }

        for (int i = 0; i < _config.terrainResources.Count; i++)
        {
            var terrain = _config.terrainResources[i];
            if (terrain == null) continue;

            EditorGUILayout.BeginVertical("Box");

            // 标题行
            EditorGUILayout.BeginHorizontal();
            EditorGUILayout.LabelField($"地形 下标{i}", EditorStyles.boldLabel);

            GUI.backgroundColor = Color.red;
            if (GUILayout.Button("删除此地形", GUILayout.Width(80)))
            {
                _config.terrainResources.RemoveAt(i);
                EditorUtility.SetDirty(_config);
                EditorGUILayout.EndHorizontal();
                EditorGUILayout.EndVertical();
                break;
            }
            GUI.backgroundColor = Color.white;
            EditorGUILayout.EndHorizontal();

            // 基础信息
            terrain.terrainId = EditorGUILayout.TextField("唯一 ID", terrain.terrainId);
            terrain.terrainName = EditorGUILayout.TextField("显示名称", terrain.terrainName);
            terrain.terrainSprite = EditorGUILayout.ObjectField(
                "预览图标", terrain.terrainSprite, typeof(Sprite), false) as Sprite;

            // 通行规则
            terrain.defaultType = (DefaultTerrainType)EditorGUILayout.EnumPopup(
                "通行规则", terrain.defaultType);

            // 特定单位阻挡列表
            if (terrain.defaultType == DefaultTerrainType.UnitImpassable)
            {
                EditorGUILayout.LabelField("禁止通行的单位 ID 列表");
                EditorGUI.indentLevel++;
                for (int j = 0; j < terrain.blockUnitIds.Count; j++)
                {
                    EditorGUILayout.BeginHorizontal();
                    terrain.blockUnitIds[j] = EditorGUILayout.TextField(
                        $"单位 {j}", terrain.blockUnitIds[j]);
                    if (GUILayout.Button("×", GUILayout.Width(25)))
                    {
                        terrain.blockUnitIds.RemoveAt(j);
                        EditorUtility.SetDirty(_config);
                    }
                    EditorGUILayout.EndHorizontal();
                }
                if (GUILayout.Button("+ 添加禁止单位 ID"))
                {
                    terrain.blockUnitIds.Add("");
                    EditorUtility.SetDirty(_config);
                }
                EditorGUI.indentLevel--;
            }

            // 自定义属性
            EditorGUILayout.LabelField("自定义属性");
            EditorGUI.indentLevel++;

            for (int j = 0; j < terrain.customProperties.Count; j++)
            {
                var prop = terrain.customProperties[j];
                if (prop == null) continue;

                EditorGUILayout.BeginVertical("Box");

                EditorGUILayout.BeginHorizontal();
                prop.propertyName = EditorGUILayout.TextField("属性名", prop.propertyName);
                prop.propertyType = (PropertyType)EditorGUILayout.EnumPopup(prop.propertyType, GUILayout.Width(60));
                if (GUILayout.Button("×", GUILayout.Width(25)))
                {
                    terrain.customProperties.RemoveAt(j);
                    EditorUtility.SetDirty(_config);
                    EditorGUILayout.EndHorizontal();
                    EditorGUILayout.EndVertical();
                    break;
                }
                EditorGUILayout.EndHorizontal();

                switch (prop.propertyType)
                {
                    case PropertyType.Int:
                        prop.intValue = EditorGUILayout.IntField("数值", prop.intValue);
                        break;
                    case PropertyType.Float:
                        prop.floatValue = EditorGUILayout.FloatField("数值", prop.floatValue);
                        break;
                    case PropertyType.Bool:
                        prop.boolValue = EditorGUILayout.Toggle("是否启用", prop.boolValue);
                        break;
                }

                prop.propertyEffect = EditorGUILayout.TextArea(
                    prop.propertyEffect, GUILayout.Height(36));

                EditorGUILayout.EndVertical();
            }

            if (GUILayout.Button("+ 添加自定义属性"))
            {
                terrain.customProperties.Add(new CustomProperty());
                EditorUtility.SetDirty(_config);
            }

            EditorGUI.indentLevel--;
            EditorGUILayout.EndVertical();
            EditorGUILayout.Space();
        }

        EditorGUILayout.EndScrollView();
    }

    // ==================== Scene 视图 ====================

    private void OnSceneGUI(SceneView sceneView)
    {
        if (_config == null) return;

        DrawGridGizmos();
        HandleMouseInput();
    }

    /// <summary>绘制网格线与已放置的地形色块</summary>
    private void DrawGridGizmos()
    {
        float w = _mapWidth * _cellSize;
        float h = _mapHeight * _cellSize;

        // 网格线
        Handles.color = new Color(0.2f, 0.8f, 0.2f, 0.3f);
        for (int x = 0; x <= _mapWidth; x++)
        {
            Vector3 start = new Vector3(x * _cellSize, 0, 0);
            Vector3 end = new Vector3(x * _cellSize, 0, h);
            Handles.DrawLine(start, end);
        }
        for (int y = 0; y <= _mapHeight; y++)
        {
            Vector3 start = new Vector3(0, 0, y * _cellSize);
            Vector3 end = new Vector3(w, 0, y * _cellSize);
            Handles.DrawLine(start, end);
        }

        // 已放置地形色块
        foreach (var kv in _mapGrid)
        {
            Vector2Int cell = kv.Key;
            string tid = kv.Value;
            var terrain = _config.GetTerrainById(tid);

            Color col = Color.gray;
            if (terrain != null)
            {
                switch (terrain.defaultType)
                {
                    case DefaultTerrainType.Passable:
                        col = Color.green;
                        break;
                    case DefaultTerrainType.Impassable:
                        col = Color.red;
                        break;
                    case DefaultTerrainType.UnitImpassable:
                        col = Color.yellow;
                        break;
                }
            }

            Handles.color = new Color(col.r, col.g, col.b, 0.4f);
            Vector3 center = new Vector3(
                cell.x * _cellSize + _cellSize * 0.5f,
                0,
                cell.y * _cellSize + _cellSize * 0.5f);

            Handles.CubeHandleCap(0, center, Quaternion.identity,
                _cellSize * 0.9f, EventType.Repaint);
        }
    }

    /// <summary>处理鼠标点击输入（刷子/橡皮）</summary>
    private void HandleMouseInput()
    {
        Event e = Event.current;
        if (e == null) return;

        Ray ray = HandleUtility.GUIPointToWorldRay(e.mousePosition);
        Plane plane = new Plane(Vector3.up, Vector3.zero);

        if (!plane.Raycast(ray, out float dist)) return;

        Vector3 worldPos = ray.GetPoint(dist);
        int cellX = Mathf.FloorToInt(worldPos.x / _cellSize);
        int cellY = Mathf.FloorToInt(worldPos.z / _cellSize);

        // 边界检查
        if (cellX < 0 || cellX >= _mapWidth || cellY < 0 || cellY >= _mapHeight)
            return;

        Vector2Int cellPos = new Vector2Int(cellX, cellY);

        // 只在左键点击时处理
        if (e.type != EventType.MouseDown || e.button != 0)
            return;

        if (_editMode == EditMode.Brush)
        {
            if (_selectTerrainIndex < _config.terrainResources.Count)
            {
                string tid = _config.terrainResources[_selectTerrainIndex].terrainId;
                _mapGrid[cellPos] = tid; // 添加或覆写
                Repaint(); // 刷新窗口
            }
        }
        else if (_editMode == EditMode.Eraser)
        {
            _mapGrid.Remove(cellPos);
            Repaint();
        }

        e.Use();
    }

    // ==================== JSON 导入导出 ====================

    private void SaveMapToJson()
    {
        MapJsonData data = new MapJsonData
        {
            mapName = _mapName,
            mapWidth = _mapWidth,
            mapHeight = _mapHeight
        };

        foreach (var kv in _mapGrid)
        {
            data.cellDatas.Add(new MapCellData
            {
                x = kv.Key.x,
                y = kv.Key.y,
                terrainId = kv.Value
            });
        }

        string path = EditorUtility.SaveFilePanel(
            "保存地图 JSON", "Assets", _mapName, "json");

        if (string.IsNullOrEmpty(path)) return;

        string json = JsonUtility.ToJson(data, prettyPrint: true);
        File.WriteAllText(path, json);
        AssetDatabase.Refresh();

        EditorUtility.DisplayDialog("保存成功", $"地图已保存到:\n{path}", "确定");
    }

    private void LoadMapFromJson()
    {
        string path = EditorUtility.OpenFilePanel(
            "选择地图 JSON", "Assets", "json");

        if (string.IsNullOrEmpty(path)) return;

        string json = File.ReadAllText(path);
        MapJsonData data = JsonUtility.FromJson<MapJsonData>(json);

        _mapName = data.mapName;
        _mapWidth = data.mapWidth;
        _mapHeight = data.mapHeight;
        _mapGrid.Clear();

        foreach (var cell in data.cellDatas)
        {
            _mapGrid[new Vector2Int(cell.x, cell.y)] = cell.terrainId;
        }

        Repaint();
        EditorUtility.DisplayDialog(
            "加载成功",
            $"已加载地图: {data.mapName}\n尺寸: {data.mapWidth}×{data.mapHeight}\n地形格子: {data.cellDatas.Count}",
            "确定");
    }
}
```

### 第一步：初始化环境

1. 按上方目录结构创建文件夹，将对应脚本粘贴到对应位置
2. 右键 Project 窗口 → **Create → 地图编辑器 → 地形资源配置表**，创建 `.asset` 配置文件
3. 顶部菜单 → **地图编辑器 → 打开地图编辑器**，打开编辑器窗口

### 第二步：配置地形范例

在编辑器中配置 3 种示例地形：

#### 地形1 — 草地（可通行）

| 字段 | 值 |
|------|-----|
| 唯一 ID | `grass` |
| 通行规则 | `Passable`（可通过） |
| 自定义属性 | 攻击加成 Int = 1，移速加成 Float = 0.2 |

效果：站在草地时攻击 +1，移速提升 20%

#### 地形2 — 山石（全体不可通行）

| 字段 | 值 |
|------|-----|
| 唯一 ID | `stone` |
| 通行规则 | `Impassable`（不可通过） |
| 自定义属性 | 无 |

效果：任何单位都无法通过

#### 地形3 — 浅滩（指定单位不可通行）

| 字段 | 值 |
|------|-----|
| 唯一 ID | `shoal` |
| 通行规则 | `UnitImpassable`（指定单位不可通过） |
| 禁止通行单位 ID | `infantry`（步兵） |
| 自定义属性 | 是否中毒 Bool = true |

效果：步兵无法通过浅滩；其他单位可通过但中毒掉血

### 第三步：编辑地图

1. 设置地图宽度、高度、格子尺寸
2. 编辑模式切换为 **Brush（刷子）**
3. 通过滑块选择要绘制的地形下标
4. 在 **Scene 视图** 中，鼠标左键点击格子即可放置地形
5. 切换到 **Eraser（橡皮）** 可擦除已放置的格子
6. 颜色指示：
   - **绿色** = 可通过
   - **红色** = 阻挡
   - **黄色** = 特定单位阻挡

### 第四步：保存 & 加载 JSON

- 点击 **保存地图为 JSON** → 选择保存路径 → 生成 JSON 文件

### 扩展1：运行时地图生成器

路径：`Assets/Scripts/Map/RuntimeMapGenerator.cs`

在游戏启动时从 JSON 自动生成完整地图（含地形 Sprite、碰撞、交互触发器等）：

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 运行时地图生成器。
/// 挂载到场景空物体上，拖入配置和 JSON 文件，运行时自动构建地图。
/// </summary>
public class RuntimeMapGenerator : MonoBehaviour
{
    [Header("配置")]
    public MapResourceConfig config;           // 地形资源配置表
    public TextAsset mapJson;                  // 地图 JSON 文件
    public Transform mapRoot;                  // 地图父节点（不填则自动创建）

    [Header("渲染")]
    public Material terrainMaterial;           // 地形材质（可选）
    public float spriteYOffset = 0.01f;        // Sprite 高度偏移，避免 Z-fighting

    [Header("碰撞")]
    public bool generateColliders = true;      // 是否为阻挡地形生成碰撞体
    public PhysicMaterial blockPhysicMaterial; // 阻挡地形物理材质

    // —— 运行时状态 ——
    private Dictionary<Vector2Int, GameObject> _spawnedCells = new Dictionary<Vector2Int, GameObject>();
    private MapJsonData _loadedData;

    private void Awake()
    {
        GenerateMap();
    }

    /// <summary>生成 / 重建整个地图</summary>
    [ContextMenu("生成地图")]
    public void GenerateMap()
    {
        ClearMap();

        // 初始化全局配置
        GameMapGlobal.MapConfig = config;

        if (config == null)
        {
            Debug.LogError("[RuntimeMapGenerator] 地形资源配置表为空！");
            return;
        }
        if (mapJson == null)
        {
            Debug.LogError("[RuntimeMapGenerator] 地图 JSON 文件为空！");
            return;
        }

        _loadedData = JsonUtility.FromJson<MapJsonData>(mapJson.text);
        if (_loadedData == null)
        {
            Debug.LogError("[RuntimeMapGenerator] JSON 解析失败！");
            return;
        }

        if (mapRoot == null)
        {
            mapRoot = new GameObject("RuntimeMap").transform;
            mapRoot.SetParent(transform);
        }

        // 创建背景底板
        CreateMapFloor();

        // 按格子生成
        foreach (var cell in _loadedData.cellDatas)
        {
            SpawnCell(cell);
        }

        Debug.Log($"[RuntimeMapGenerator] 地图生成完毕：" +
                  $"尺寸 {_loadedData.mapWidth}×{_loadedData.mapHeight}, " +
                  $"格子数 {_loadedData.cellDatas.Count}");
    }

    /// <summary>生成底板（半透明大平面）</summary>
    private void CreateMapFloor()
    {
        GameObject floor = GameObject.CreatePrimitive(PrimitiveType.Plane);
        floor.name = "MapFloor";
        floor.transform.SetParent(mapRoot);

        // Unity 默认 Plane 是 10×10，缩放到地图尺寸
        float scaleX = _loadedData.mapWidth / 10f;
        float scaleZ = _loadedData.mapHeight / 10f;
        floor.transform.localScale = new Vector3(scaleX, 1f, scaleZ);
        floor.transform.position = new Vector3(
            _loadedData.mapWidth * 0.5f, -0.01f, _loadedData.mapHeight * 0.5f);

        // 半透明灰色底板
        var renderer = floor.GetComponent<MeshRenderer>();
        if (terrainMaterial != null)
            renderer.material = terrainMaterial;
        else
        {
            var mat = new Material(Shader.Find("Standard"));
            mat.color = new Color(0.3f, 0.3f, 0.3f, 0.5f);
            mat.SetFloat("_Mode", 3); // Transparent
            renderer.material = mat;
        }

        DestroyImmediate(floor.GetComponent<Collider>()); // 底板不需要碰撞
    }

    /// <summary>生成单个格子</summary>
    private void SpawnCell(MapCellData cell)
    {
        var terrain = config.GetTerrainById(cell.terrainId);
        if (terrain == null)
        {
            Debug.LogWarning($"[RuntimeMapGenerator] 未知地形 ID: {cell.terrainId}");
            return;
        }

        Vector3 pos = new Vector3(cell.x + 0.5f, spriteYOffset, cell.y + 0.5f);
        GameObject cellGo = new GameObject($"Cell_{cell.x}_{cell.y}");
        cellGo.transform.SetParent(mapRoot);
        cellGo.transform.position = pos;

        // Sprite 渲染
        if (terrain.terrainSprite != null)
        {
            var sr = cellGo.AddComponent<SpriteRenderer>();
            sr.sprite = terrain.terrainSprite;
            sr.sortingOrder = 0;

            // 根据通行类型染色（可选）
            switch (terrain.defaultType)
            {
                case DefaultTerrainType.Passable:
                    sr.color = new Color(1f, 1f, 1f, 0.9f);
                    break;
                case DefaultTerrainType.Impassable:
                    sr.color = new Color(1f, 0.7f, 0.7f, 0.9f);
                    break;
                case DefaultTerrainType.UnitImpassable:
                    sr.color = new Color(1f, 1f, 0.7f, 0.9f);
                    break;
            }
        }

        // 碰撞体（阻挡地形用）
        if (generateColliders && terrain.defaultType == DefaultTerrainType.Impassable)
        {
            var col = cellGo.AddComponent<BoxCollider>();
            col.size = new Vector3(1f, 1f, 1f);
            if (blockPhysicMaterial != null)
                col.material = blockPhysicMaterial;
        }

        // 地形数据组件（运行时查询用）
        var cellData = cellGo.AddComponent<TerrainCellData>();
        cellData.terrainId = cell.terrainId;
        cellData.gridX = cell.x;
        cellData.gridY = cell.y;

        _spawnedCells[new Vector2Int(cell.x, cell.y)] = cellGo;
    }

    /// <summary>清除已生成的地图</summary>
    [ContextMenu("清除地图")]
    public void ClearMap()
    {
        if (mapRoot != null)
        {
            for (int i = mapRoot.childCount - 1; i >= 0; i--)
            {
                Destroy(mapRoot.GetChild(i).gameObject);
            }
        }
        _spawnedCells.Clear();
    }

    /// <summary>获取已生成的格子（外部调用）</summary>
    public GameObject GetCell(int x, int y)
    {
        _spawnedCells.TryGetValue(new Vector2Int(x, y), out var go);
        return go;
    }
}

/// <summary>挂载到每个生成格子上的运行时数据组件</summary>
public class TerrainCellData : MonoBehaviour
{
    public string terrainId;
    public int gridX;
    public int gridY;

    /// <summary>获取当前格子的地形资源</summary>
    public TerrainResource GetTerrain()
    {
        return GameMapGlobal.GetTerrain(terrainId);
    }

    /// <summary>单位进入此格子时调用</summary>
    public void OnUnitEnter(GameObject unit)
    {
        var logic = unit.GetComponent<UnitTerrainLogic>();
        if (logic != null)
            logic.OnEnterGrid(terrainId);
    }
}
```

```csharp
    // ===== 多层地图查询（扩展） =====

    /// <summary>按层+坐标获取地形资源</summary>
    public static TerrainResource GetTerrainAt(int x, int y, int layer, MapJsonData mapData)
    {
        if (mapData == null) return null;
        foreach (var cell in mapData.cellDatas)
        {
            if (cell.x == x && cell.y == y && cell.layer == layer)
                return GetTerrain(cell.terrainId);
        }
        return null;
    }

    /// <summary>判断某层某坐标是否可为某单位通过</summary>
    public static bool CheckUnitCanPassAt(int x, int y, int layer, string unitId, MapJsonData mapData)
    {
        var terrain = GetTerrainAt(x, y, layer, mapData);
        if (terrain == null) return true;
        return CheckUnitCanPass(terrain.terrainId, unitId);
    }

    /// <summary>获取某坐标所有层中优先级最高的阻挡层（用于决定单位行走层）</summary>
    public static int GetHighestBlockingLayer(int x, int y, string unitId, MapJsonData mapData)
    {
        if (mapData == null) return 0;
        for (int layer = 0; layer < mapData.layerCount; layer++)
        {
            if (!CheckUnitCanPassAt(x, y, layer, unitId, mapData))
                return layer;
        }
        return mapData.layerCount - 1; // 全部可通行，走最顶层
    }
```

### 扩展3：A* 寻路集成

路径：`Assets/Scripts/Map/MapPathfinding.cs`

完整 A* 寻路实现，直接读取地图数据生成寻路网格：

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 基于地图数据的 A* 寻路。
/// 传入地图 JSON 数据和单位 ID，自动考虑地形通行规则。
/// </summary>
public static class MapPathfinding
{
    /// <summary>查找从起点到终点的最短路径</summary>
    /// <param name="startX">起点 X 坐标</param>
    /// <param name="startY">起点 Y 坐标</param>
    /// <param name="endX">终点 X 坐标</param>
    /// <param name="endY">终点 Y 坐标</param>
    /// <param name="unitId">单位 ID（用于通行判断）</param>
    /// <param name="mapData">地图数据</param>
    /// <param name="allowDiagonal">是否允许斜向移动</param>
    /// <returns>路径节点列表（不含起点），空列表表示无路径</returns>
    public static List<Vector2Int> FindPath(
        int startX, int startY,
        int endX, int endY,
        string unitId,
        MapJsonData mapData,
        bool allowDiagonal = false)
    {
        // 边界检查
        if (mapData == null) return new List<Vector2Int>();
        if (!IsInBounds(startX, startY, mapData) || !IsInBounds(endX, endY, mapData))
            return new List<Vector2Int>();

        // 终点不可通行
        if (!IsPassable(endX, endY, unitId, mapData))
            return new List<Vector2Int>();

        Vector2Int start = new Vector2Int(startX, startY);
        Vector2Int end = new Vector2Int(endX, endY);

        // Open / Closed 集合
        var openSet = new Heap<Node>(mapData.mapWidth * mapData.mapHeight);
        var closedSet = new HashSet<Vector2Int>();

        // 节点缓存（避免重复创建）
        var nodeCache = new Dictionary<Vector2Int, Node>();

        Node startNode = GetOrCreateNode(startX, startY, nodeCache);
        startNode.gCost = 0;
        startNode.hCost = Heuristic(startX, startY, endX, endY);
        openSet.Add(startNode);

        while (openSet.Count > 0)
        {
            Node current = openSet.RemoveFirst();

            if (current.pos == end)
            {
                // 构建路径
                return RetracePath(startNode, current);
            }

            closedSet.Add(current.pos);

            // 遍历邻居
            int[][] directions = allowDiagonal
                ? new int[][] { // 8 方向
                    new[] { 0, 1 }, new[] { 1, 0 }, new[] { 0, -1 }, new[] { -1, 0 },
                    new[] { 1, 1 }, new[] { 1, -1 }, new[] { -1, -1 }, new[] { -1, 1 } }
                : new int[][] { // 4 方向
                    new[] { 0, 1 }, new[] { 1, 0 }, new[] { 0, -1 }, new[] { -1, 0 } };

            foreach (var dir in directions)
            {
                int nx = current.pos.x + dir[0];
                int ny = current.pos.y + dir[1];

                if (!IsInBounds(nx, ny, mapData)) continue;
                if (closedSet.Contains(new Vector2Int(nx, ny))) continue;
                if (!IsPassable(nx, ny, unitId, mapData)) continue;

                // 斜向移动时检查两边是否都是可通行的（防止穿角）
                if (Mathf.Abs(dir[0]) == 1 && Mathf.Abs(dir[1]) == 1)
                {
                    if (!IsPassable(current.pos.x + dir[0], current.pos.y, unitId, mapData)) continue;
                    if (!IsPassable(current.pos.x, current.pos.y + dir[1], unitId, mapData)) continue;
                }

                Node neighbor = GetOrCreateNode(nx, ny, nodeCache);
                float moveCost = (Mathf.Abs(dir[0]) + Mathf.Abs(dir[1]) == 2) ? 1.414f : 1f;

                // 考虑地形额外消耗（如果有"移动消耗"自定义属性）
                var terrain = GameMapGlobal.GetTerrainAt(nx, ny, 0, mapData);
                if (terrain != null)
                {
                    float terrainCost = terrain.GetFloat("移动消耗");
                    if (terrainCost > 0) moveCost += terrainCost;
                }

                float newGCost = current.gCost + moveCost;
                if (newGCost < neighbor.gCost)
                {
                    neighbor.gCost = newGCost;
                    neighbor.hCost = Heuristic(nx, ny, endX, endY);
                    neighbor.parent = current;

                    if (!openSet.Contains(neighbor))
                        openSet.Add(neighbor);
                    else
                        openSet.UpdateItem(neighbor);
                }
            }
        }

        // 无路径
        return new List<Vector2Int>();
    }

    /// <summary>检查格子是否可通行</summary>
    public static bool IsPassable(int x, int y, string unitId, MapJsonData mapData)
    {
        // 先查第 0 层（地面层）是否可通过
        return GameMapGlobal.CheckUnitCanPassAt(x, y, 0, unitId, mapData);
    }

    private static bool IsInBounds(int x, int y, MapJsonData mapData)
    {
        return x >= 0 && x < mapData.mapWidth && y >= 0 && y < mapData.mapHeight;
    }

    private static int Heuristic(int x1, int y1, int x2, int y2)
    {
        // 曼哈顿距离（适合 4 方向移动）或 八方向用对角距离
        int dx = Mathf.Abs(x1 - x2);
        int dy = Mathf.Abs(y1 - y2);
        // 切比雪夫距离（八方向）
        return 10 * (dx + dy) + (14 - 2 * 10) * Mathf.Min(dx, dy);
    }

    private static List<Vector2Int> RetracePath(Node startNode, Node endNode)
    {
        List<Vector2Int> path = new List<Vector2Int>();
        Node current = endNode;
        while (current != startNode)
        {
            path.Add(current.pos);
            current = current.parent;
        }
        path.Reverse();
        return path;
    }

    private static Node GetOrCreateNode(int x, int y, Dictionary<Vector2Int, Node> cache)
    {
        var key = new Vector2Int(x, y);
        if (!cache.TryGetValue(key, out Node node))
        {
            node = new Node { pos = key };
            cache[key] = node;
        }
        return node;
    }

    // ===== 内部数据结构 =====

    private class Node : IHeapItem<Node>
    {
        public Vector2Int pos;
        public float gCost;
        public float hCost;
        public Node parent;
        public int heapIndex;

        public float fCost => gCost + hCost;

        public int CompareTo(Node other)
        {
            int compare = fCost.CompareTo(other.fCost);
            if (compare == 0)
                compare = hCost.CompareTo(other.hCost);
            return compare;
        }
    }
}

/// <summary>最小堆（用于优化 Open 集合性能）</summary>
public class Heap<T> where T : IHeapItem<T>
{
    private T[] _items;
    private int _count;

    public int Count => _count;

    public Heap(int maxSize) { _items = new T[maxSize]; }

    public void Add(T item)
    {
        item.heapIndex = _count;
        _items[_count] = item;
        SortUp(item);
        _count++;
    }

    public T RemoveFirst()
    {
        T first = _items[0];
        _count--;
        _items[0] = _items[_count];
        _items[0].heapIndex = 0;
        SortDown(_items[0]);
        return first;
    }

    public bool Contains(T item) => Equals(_items[item.heapIndex], item);

    public void UpdateItem(T item) => SortUp(item);

    private void SortUp(T item)
    {
        int parent = (item.heapIndex - 1) / 2;
        while (true)
        {
            T parentItem = _items[parent];
            if (item.CompareTo(parentItem) < 0)
                Swap(item, parentItem);
            else
                break;
            parent = (item.heapIndex - 1) / 2;
        }
    }

    private void SortDown(T item)
    {
        while (true)
        {
            int left = item.heapIndex * 2 + 1;
            int right = item.heapIndex * 2 + 2;
            int swap = left;

            if (left >= _count) break;
            if (right < _count && _items[left].CompareTo(_items[right]) > 0)
                swap = right;
            if (item.CompareTo(_items[swap]) > 0)
                Swap(item, _items[swap]);
            else
                break;
        }
    }

    private void Swap(T a, T b)
    {
        _items[a.heapIndex] = b;
        _items[b.heapIndex] = a;
        (a.heapIndex, b.heapIndex) = (b.heapIndex, a.heapIndex);
    }
}

public interface IHeapItem<T> : System.IComparable<T>
{
    int heapIndex { get; set; }
}
```

### 扩展4：拖拽连续绘制

修改 `MapEditorWindow.cs` 的 `HandleMouseInput()` 方法，支持按住左键拖拽连续绘制，并添加 Shift 按住时仅删除的快捷操作：

```csharp
// 在 MapEditorWindow 类顶部增加字段
    private Vector2Int _lastPaintedCell = new Vector2Int(int.MinValue, int.MinValue);
    private bool _isPainting = false;

// 替换原有的 HandleMouseInput 方法
    /// <summary>处理鼠标输入（支持拖拽连续绘制 + Shift 快捷擦除）</summary>
    private void HandleMouseInput()
    {
        Event e = Event.current;
        if (e == null) return;

        Ray ray = HandleUtility.GUIPointToWorldRay(e.mousePosition);
        Plane plane = new Plane(Vector3.up, Vector3.zero);

        if (!plane.Raycast(ray, out float dist)) return;

        Vector3 worldPos = ray.GetPoint(dist);
        int cellX = Mathf.FloorToInt(worldPos.x / _cellSize);
        int cellY = Mathf.FloorToInt(worldPos.z / _cellSize);

        if (cellX < 0 || cellX >= _mapWidth || cellY < 0 || cellY >= _mapHeight)
            return;

        Vector2Int cellPos = new Vector2Int(cellX, cellY);

        // —— 左键按下：开始绘制 ——
        if (e.type == EventType.MouseDown && e.button == 0)
        {
            _isPainting = true;
            _lastPaintedCell = cellPos;
            _undoStack.Push(new MapOperation(_mapGrid)); // 记录操作前快照（搭配 Undo 扩展）
            PaintCell(cellPos);
            e.Use();
            return;
        }

        // —— 左键拖拽：连续绘制（跳过已绘制的格子避免重复操作） ——
        if (e.type == EventType.MouseDrag && e.button == 0 && _isPainting)
        {
            if (cellPos != _lastPaintedCell)
            {
                _lastPaintedCell = cellPos;
                PaintCell(cellPos);
            }
            e.Use();
            return;
        }

        // —— 左键抬起：结束绘制 ——
        if (e.type == EventType.MouseUp && e.button == 0)
        {
            _isPainting = false;
            e.Use();
        }
    }

    /// <summary>在当前格子执行绘制/擦除操作</summary>
    private void PaintCell(Vector2Int cellPos)
    {
        // Shift 按住时临时切换为橡皮模式
        EditMode activeMode = _editMode;
        if (Event.current.shift)
            activeMode = EditMode.Eraser;

        if (activeMode == EditMode.Brush)
        {
            if (_selectTerrainIndex < _config.terrainResources.Count)
            {
                string tid = _config.terrainResources[_selectTerrainIndex].terrainId;
                _mapGrid[cellPos] = tid;
                Repaint();
            }
        }
        else if (activeMode == EditMode.Eraser)
        {
            if (_mapGrid.ContainsKey(cellPos))
            {
                _mapGrid.Remove(cellPos);
                Repaint();
            }
        }
    }
```

### 扩展5：撤销 / 重做（Undo / Redo）

在 `MapEditorWindow.cs` 中增加完整的操作历史栈，支持 Ctrl+Z / Ctrl+Y：

#### 5.1 在类顶部增加字段和数据结构

```csharp
// —— Undo/Redo 数据 ——
private struct MapOperation
{
    public Dictionary<Vector2Int, string> snapshot;

    public MapOperation(Dictionary<Vector2Int, string> current)
    {
        snapshot = new Dictionary<Vector2Int, string>(current);
    }
}

private Stack<MapOperation> _undoStack = new Stack<MapOperation>();
private Stack<MapOperation> _redoStack = new Stack<MapOperation>();
private const int MAX_HISTORY = 50; // 最大历史步数
```

#### 5.2 操作记录方法

```csharp
    /// <summary>记录当前地图快照到 Undo 栈</summary>
    private void RecordUndo()
    {
        _undoStack.Push(new MapOperation(_mapGrid));

        // 限制历史步数，超出时移除最旧记录
        if (_undoStack.Count > MAX_HISTORY)
        {
            // Stack 无法直接移除底部，用临时 List 中转
            var list = new List<MapOperation>(_undoStack);
            list.RemoveAt(list.Count - 1); // 移除最旧的（栈底）
            list.Reverse();
            _undoStack = new Stack<MapOperation>(list);
        }

        _redoStack.Clear(); // 新操作后清空重做栈
    }

    /// <summary>执行 Undo</summary>
    private void PerformUndo()
    {
        if (_undoStack.Count == 0) return;

        _redoStack.Push(new MapOperation(_mapGrid));
        var previous = _undoStack.Pop();
        _mapGrid = new Dictionary<Vector2Int, string>(previous.snapshot);
        Repaint();
        Debug.Log($"[Undo] 已撤销，Undo 剩余: {_undoStack.Count}, Redo: {_redoStack.Count}");
    }

    /// <summary>执行 Redo</summary>
    private void PerformRedo()
    {
        if (_redoStack.Count == 0) return;

        _undoStack.Push(new MapOperation(_mapGrid));
        var next = _redoStack.Pop();
        _mapGrid = new Dictionary<Vector2Int, string>(next.snapshot);
        Repaint();
        Debug.Log($"[Redo] 已重做，Undo 剩余: {_undoStack.Count}, Redo: {_redoStack.Count}");
    }
```

#### 5.3 在 OnGUI 中响应快捷键

### 扩展6：地形预览（鼠标悬停预览）

修改 `MapEditorWindow.cs` 的 Scene 视图绘制，在鼠标悬停时显示半透明预览色块：

```csharp
    // 在 MapEditorWindow 类中增加字段
    private Vector2Int _hoveredCell = new Vector2Int(int.MinValue, int.MinValue);
    private Vector2Int _previousHoveredCell = new Vector2Int(int.MinValue, int.MinValue);

    // 修改 OnSceneGUI 方法，增加预览绘制调用
    private void OnSceneGUI(SceneView sceneView)
    {
        if (_config == null) return;

        DrawGridGizmos();
        DrawHoverPreview();  // ← 新增：悬停预览
        HandleMouseInput();
    }

    /// <summary>绘制鼠标悬停预览色块</summary>
    private void DrawHoverPreview()
    {
        Event e = Event.current;
        if (e == null) return;

        Ray ray = HandleUtility.GUIPointToWorldRay(e.mousePosition);
        Plane plane = new Plane(Vector3.up, Vector3.zero);

        if (!plane.Raycast(ray, out float dist)) return;

        Vector3 worldPos = ray.GetPoint(dist);
        int cellX = Mathf.FloorToInt(worldPos.x / _cellSize);
        int cellY = Mathf.FloorToInt(worldPos.z / _cellSize);

        if (cellX < 0 || cellX >= _mapWidth || cellY < 0 || cellY >= _mapHeight)
            return;

        _hoveredCell = new Vector2Int(cellX, cellY);

        // 获取预览颜色
        Color previewColor;
        float alpha;

        EditMode activeMode = e.shift ? EditMode.Eraser : _editMode;

        if (activeMode == EditMode.Eraser)
        {
            // 橡皮模式：红色闪烁提示
            previewColor = Color.red;
            alpha = 0.35f;
        }
        else
        {
            // 刷子模式：根据选中的地形类型显示对应颜色
            if (_selectTerrainIndex < _config.terrainResources.Count)
            {
                var terrain = _config.terrainResources[_selectTerrainIndex];
                switch (terrain.defaultType)
                {
                    case DefaultTerrainType.Passable: previewColor = Color.green; break;
                    case DefaultTerrainType.Impassable: previewColor = Color.red; break;
                    case DefaultTerrainType.UnitImpassable: previewColor = Color.yellow; break;
                    default: previewColor = Color.gray; break;
                }
            }
            else
            {
                previewColor = Color.gray;
            }
            alpha = 0.3f;
        }

        // 绘制半透明预览块
        Handles.color = new Color(previewColor.r, previewColor.g, previewColor.b, alpha);
        Vector3 center = new Vector3(
            _hoveredCell.x * _cellSize + _cellSize * 0.5f,
            0.05f, // 略高于已放置的地形块
            _hoveredCell.y * _cellSize + _cellSize * 0.5f);

        Handles.CubeHandleCap(0, center, Quaternion.identity,
            _cellSize * 0.95f, EventType.Repaint);

        // 绘制边框高亮
        Handles.color = new Color(1f, 1f, 1f, 0.6f);
        Vector3[] corners = new Vector3[5]
        {
            new Vector3(cellX * _cellSize,             0.06f, cellY * _cellSize),
            new Vector3((cellX + 1) * _cellSize,       0.06f, cellY * _cellSize),
            new Vector3((cellX + 1) * _cellSize,       0.06f, (cellY + 1) * _cellSize),
            new Vector3(cellX * _cellSize,             0.06f, (cellY + 1) * _cellSize),
            new Vector3(cellX * _cellSize,             0.06f, cellY * _cellSize),
        };
        Handles.DrawAAPolyLine(2f, corners);

        // 鼠标离开格子时触发 Scene 视图刷新（消除旧预览残影）
        if (_hoveredCell != _previousHoveredCell)
        {
            _previousHoveredCell = _hoveredCell;
            SceneView.RepaintAll();
        }

        // 在鼠标旁显示地形信息提示
        Handles.BeginGUI();
        Vector2 mousePos = e.mousePosition;
        string tooltip;

        if (activeMode == EditMode.Eraser)
        {
            tooltip = _mapGrid.ContainsKey(_hoveredCell)
                ? "🗑 擦除此地形象"
                : "（空格子）";
        }
        else
        {
            var terrain = _config.terrainResources.Count > _selectTerrainIndex
                ? _config.terrainResources[_selectTerrainIndex] : null;
            tooltip = terrain != null
                ? $"🖌 {terrain.terrainName} (ID: {terrain.terrainId})"
                : "无选中的地形";
        }

        var style = new GUIStyle(GUI.skin.box)
        {
            fontSize = 12,
            alignment = TextAnchor.MiddleLeft,
            normal = { textColor = Color.white }
        };
        var content = new GUIContent(tooltip);
        Vector2 size = style.CalcSize(content);
        GUI.Box(new Rect(mousePos.x + 20, mousePos.y - 10, size.x + 10, 22), content, style);
        Handles.EndGUI();
    }
```
